#include <stdio.h>

 void trocaValores(int *n1,int *n2){
   int aux;
   aux=*n1;
   *n1=*n2;
   *n2=aux;
}

 int main(){
   int n1,n2;
   printf("N1: ");
   scanf("%d",&n1);
   printf("N2: ");
   scanf("%d",&n2);
   printf("\nEndereço de N1:%p",&n1);
   printf("\nEndereço de N2:%p",&n2);
   trocaValores(&n1,&n2);
   printf("\nN1:%d",n1);
   printf("\nN2:%d",n2);
   }
